def create_table
  transaction do
    connection.create_table(table_name, id: false)
    create_fields

    connection.execute("ALTER TABLE #{table_name} ADD PRIMARY KEY (uuid);")
    connection.add_index(table_name, :location, spatial: true)
  end
end
